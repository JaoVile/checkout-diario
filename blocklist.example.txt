#  Formatos aceitos:
#    repo:<nome>     ignora um repositório git                ex: repo:projeto-secreto

#    path:<texto>    ignora comandos/pastas que contenham     ex: path:Guinchomaps
#    user:<texto>    ignora sessões/sudo que contenham        ex: user:smartctl
#    site:<texto>    ignora sites no histórico do navegador   ex: site:youtube.com
#
#  Para bloquear na hora, use:  ~/Checkouts/bin/block.sh repo Nome
#
